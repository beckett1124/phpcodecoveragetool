481   {

500 
501     //header("Content-type: image/png");

504     //header("Content-Disposition: attachment; filename=".$file_name);
505 
506     $fp = fopen($file_path,"r");
507     $buffer_size = 1024;
508     $cur_pos = 0;
509     $result = null;
510 
511     while(!feof($fp) && ($file_size - $cur_pos > $buffer_size))

514       $result .= $buffer;

517     }
518

520     //echo $buffer;
521     $result .= $buffer;
522     fclose($fp);
523     if ($decrypted) {
524       $result = Util::decrypt_data($result);
525     }
526     return $result;
